# 快速上手

# 引入 css 样式

使用该组件，有一些CSS样式上的需求。

请引入 css 文件来实现组件的样式

import 'skyz-gulu/dist/index.css'

使用本框架前，请在 CSS 中开启 border-box

*,*::before,*::after{box-sizing: border-box;}

# 引入组件

import {Button, Icon} from 'skyz-gulu'
import 'skyz-gulu/dist/index.css'
 export default {
    name: 'app',
    components: {
      'g-button': Button,
      'g-icon': Icon
    }
  }
